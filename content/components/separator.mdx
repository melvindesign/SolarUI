### Manual

**1. Install the following dependency:**

```bash filename="terminal" copy
npm install @radix-ui/react-separator
```

**2. Copy and paste the following code into your project:**

```tsx filename="components/ui/separator.tsx" copy
"use client"

import { Separator as SeparatorPrimitive } from "radix-ui"
import * as React from "react"

import { cn } from "@/lib/utils"

function Separator({
  className,
  orientation = "horizontal",
  decorative = true,
  ...props
}: React.ComponentProps<typeof SeparatorPrimitive.Root>) {
  return (
    <SeparatorPrimitive.Root
      data-slot="separator"
      decorative={decorative}
      orientation={orientation}
      className={cn(
        "shrink-0 bg-default-6 data-[orientation=horizontal]:h-px data-[orientation=horizontal]:w-full data-[orientation=vertical]:w-px data-[orientation=vertical]:self-stretch",
        className
      )}
      {...props}
    />
  )
}

export { Separator }
```

### Separator

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | The axis along which the separator is drawn. |
| `decorative` | `boolean` | `true` | When `true`, the separator is purely visual and hidden from assistive technologies. |
| `className` | `string` | — | Additional CSS classes to merge. |
