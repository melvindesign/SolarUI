### Manual

**1. Install the following dependency:**

```bash filename="terminal" copy
npm install @radix-ui/react-separator
```

**2. Copy and paste the following code into your project:**

```tsx filename="components/ui/separator.tsx" copy
"use client"

import * as React from "react"
import * as SeparatorPrimitive from "@radix-ui/react-separator"
import { cn } from "@/lib/utils"

const Separator = React.forwardRef<
  React.ElementRef<typeof SeparatorPrimitive.Root>,
  React.ComponentPropsWithoutRef<typeof SeparatorPrimitive.Root>
>(
  ({ className, orientation = "horizontal", decorative = true, ...props }, ref) => (
    <SeparatorPrimitive.Root
      ref={ref}
      decorative={decorative}
      orientation={orientation}
      className={cn(
        "shrink-0 bg-border",
        orientation === "horizontal" ? "h-[1px] w-full" : "h-full w-[1px]",
        className
      )}
      {...props}
    />
  )
)
Separator.displayName = SeparatorPrimitive.Root.displayName

export { Separator }
```

### Separator

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | The axis along which the separator is drawn. |
| `decorative` | `boolean` | `true` | When `true`, the separator is purely visual and hidden from assistive technologies. |
| `className` | `string` | — | Additional CSS classes to merge. |
