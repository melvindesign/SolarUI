### Manual

**1. Install the following dependency:**

```bash filename="terminal" copy
npm install @radix-ui/react-toggle
```

**2. Copy and paste the following code into your project:**

```tsx filename="components/ui/toggle.tsx" copy
"use client"

import { cva, type VariantProps } from "class-variance-authority"
import { Toggle as TogglePrimitive } from "radix-ui"
import * as React from "react"

import { cn } from "@/lib/utils"

const toggleVariants = cva(
  "hover:text-default-12 aria-pressed:bg-default-3 focus-visible:border-default-8 focus-visible:ring-default-7 aria-invalid:ring-error-7 aria-invalid:border-error-8 data-[state=on]:bg-default-3 gap-1 rounded-lg text-sm font-medium transition-all [&_svg:not([class*='size-'])]:size-4 group/toggle inline-flex items-center justify-center whitespace-nowrap outline-none hover:bg-default-3 focus-visible:ring-[3px] disabled:pointer-events-none disabled:opacity-50 [&_svg]:pointer-events-none [&_svg]:shrink-0",
  {
    variants: {
      variant: {
        default: "bg-transparent",
        outline: "border-default-7 hover:bg-default-3 border bg-transparent",
      },
      size: {
        default: "h-8 min-w-8 px-2",
        compact: "h-7 min-w-7 px-1.5 text-xs",
      },
    },
    defaultVariants: {
      variant: "default",
      size: "default",
    },
  }
)

function Toggle({
  className,
  variant = "default",
  size = "default",
  ...props
}: React.ComponentProps<typeof TogglePrimitive.Root> &
  VariantProps<typeof toggleVariants>) {
  return (
    <TogglePrimitive.Root
      data-slot="toggle"
      className={cn(toggleVariants({ variant, size, className }))}
      {...props}
    />
  )
}

export { Toggle, toggleVariants }
```

### Toggle

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `pressed` | `boolean` | — | Controlled pressed state. |
| `defaultPressed` | `boolean` | `false` | Initial pressed state (uncontrolled). |
| `onPressedChange` | `(pressed: boolean) => void` | — | Callback when pressed state changes. |
| `disabled` | `boolean` | `false` | Prevents the toggle from being interacted with. |
| `variant` | `"default" \| "outline"` | `"default"` | The visual style of the toggle. |
| `size` | `"default" \| "compact"` | `"default"` | The size of the toggle. |
| `aria-label` | `string` | — | Accessible label, required when the toggle contains only an icon. |
