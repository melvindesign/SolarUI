### Manual

Copy and paste the following code into your project:

```tsx filename="components/ui/title.tsx" copy
import { cva, type VariantProps } from "class-variance-authority"
import { Slot } from "radix-ui"
import * as React from "react"

import { cn } from "@/lib/utils"

const titleVariants = cva(
  "font-sans text-default-12",
  {
    variants: {
      variant: {
        "1": "text-title-1 tracking-display-tight font-semibold",
        "2": "text-title-2 tracking-display-tight font-semibold",
        "3": "text-title-3 tracking-display-normal font-medium",
        "4": "text-title-4 tracking-display-normal font-medium",
        "5": "text-title-5 tracking-display-loose font-medium",
        "6": "text-title-6 tracking-display-loose font-medium",
      },
    },
    defaultVariants: {
      variant: "2",
    },
  }
)

const variantTagMap: Record<string, keyof React.JSX.IntrinsicElements> = {
  "1": "h1", "2": "h2", "3": "h3", "4": "h4", "5": "h5", "6": "h6",
}

type TitleProps = React.HTMLAttributes<HTMLHeadingElement> &
  VariantProps<typeof titleVariants> & { asChild?: boolean }

function Title({ className, variant = "2", asChild = false, ...props }: TitleProps) {
  const Tag = asChild ? Slot.Root : variantTagMap[variant ?? "2"]
  return (
    <Tag
      data-slot="title"
      data-size={variant}
      className={cn(titleVariants({ size }), className)}
      {...props}
    />
  )
}

export { Title, titleVariants }
```

### Title

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `"1" \| "2" \| "3" \| "4" \| "5" \| "6"` | `"2"` | Controls the size and renders as the matching heading tag (`h1`–`h6`). |
| `asChild` | `boolean` | `false` | Merges props onto the child element instead of rendering a heading. |
| `className` | `string` | — | Additional CSS classes to merge. |

The `Title` component also accepts all standard HTML heading attributes.
