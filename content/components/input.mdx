**1. Copy and paste the following code into your project:**

```tsx filename="components/ui/input.tsx" copy
import * as React from "react"
import { cn } from "@/lib/utils"

export interface InputProps extends React.InputHTMLAttributes<HTMLInputElement> {}

const Input = React.forwardRef<HTMLInputElement, InputProps>(
  ({ className, type, ...props }, ref) => {
    return (
      <input
        type={type}
        className={cn(
          "flex h-10 w-full rounded-md border border-input bg-background px-3 py-2 text-sm",
          "placeholder:text-muted-foreground",
          "file:border-0 file:bg-transparent file:text-sm file:font-medium",
          "focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2",
          "disabled:cursor-not-allowed disabled:opacity-50",
          className
        )}
        ref={ref}
        {...props}
      />
    )
  }
)
Input.displayName = "Input"

export { Input }
```

### Input

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `string` | `"text"` | The type of the input (e.g. `text`, `email`, `password`, `file`). |
| `placeholder` | `string` | — | Placeholder text shown when the field is empty. |
| `disabled` | `boolean` | `false` | When `true`, prevents user interaction and applies reduced opacity. |
| `className` | `string` | — | Additional CSS classes to merge. |

The `Input` component accepts all standard HTML `<input>` attributes.
