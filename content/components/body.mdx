### Manual

Copy and paste the following code into your project:

```tsx filename="components/ui/body.tsx" copy
import { cva, type VariantProps } from "class-variance-authority"
import { Slot } from "radix-ui"
import * as React from "react"

import { cn } from "@/lib/utils"

const bodyVariants = cva(
  "font-sans text-default-12",
  {
    variants: {
      variant: {
        default: "text-body tracking-body font-normal",
        compact: "text-body-compact tracking-body font-normal",
      },
    },
    defaultVariants: {
      variant: "default",
    },
  }
)

type BodyProps = React.HTMLAttributes<HTMLParagraphElement> &
  VariantProps<typeof bodyVariants> & { asChild?: boolean }

function Body({ className, variant = "default", asChild = false, ...props }: BodyProps) {
  const Comp = asChild ? Slot.Root : "p"
  return (
    <Comp
      data-slot="body"
      data-variant={variant}
      className={cn(bodyVariants({ variant }), className)}
      {...props}
    />
  )
}

export { Body, bodyVariants }
```

### Body

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"default" \| "small"` | `"default"` | `default` is 16px, `small` is 14.22px. |
| `asChild` | `boolean` | `false` | Merges props onto the child element instead of rendering a `<p>`. |
| `className` | `string` | — | Additional CSS classes to merge. |

The `Body` component also accepts all standard HTML paragraph attributes.
