### Manual

Copy and paste the following code into your project:

```tsx filename="components/ui/label.tsx" copy
"use client"

import { Label as LabelPrimitive } from "radix-ui"
import * as React from "react"
import { cva, type VariantProps } from "class-variance-authority"

import { cn } from "@/lib/utils"

const labelVariants = cva(
  "font-sans font-medium text-default-12 tracking-label flex items-center gap-2 select-none group-data-[disabled=true]:opacity-50 peer-disabled:opacity-50 group-data-[disabled=true]:pointer-events-none peer-disabled:cursor-not-allowed",
  {
    variants: {
      variant: {
        default: "text-label",
        compact: "text-label-compact",
      },
    },
    defaultVariants: {
      variant: "default",
    },
  }
)

type LabelProps = React.ComponentProps<typeof LabelPrimitive.Root> &
  VariantProps<typeof labelVariants>

function Label({ className, variant, ...props }: LabelProps) {
  return (
    <LabelPrimitive.Root
      data-slot="label"
      className={cn(labelVariants({ size }), className)}
      {...props}
    />
  )
}

export { Label, labelVariants }
```

### Label

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `"default" \| "compact"` | `"default"` | `default` is 14.22px, `compact` is 12.44px. |
| `htmlFor` | `string` | — | Associates the label with a form control by its `id`. |
| `className` | `string` | — | Additional CSS classes to merge. |

The `Label` component also accepts all standard HTML `<label>` attributes.
