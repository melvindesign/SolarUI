### Manual

Copy and paste the following code into your project:

```tsx filename="components/ui/label.tsx" copy
import * as React from "react"
import { cn } from "@/lib/utils"

const Label = React.forwardRef<
  HTMLLabelElement,
  React.LabelHTMLAttributes<HTMLLabelElement> & { required?: boolean }
>(({ className, required, children, ...props }, ref) => (
  <label
    ref={ref}
    className={cn(
      "text-sm font-medium leading-none peer-disabled:cursor-not-allowed peer-disabled:opacity-70",
      className
    )}
    {...props}
  >
    {children}
    {required && <span className="text-(--color-destructive) ml-1">*</span>}
  </label>
))
Label.displayName = "Label"

export { Label }
```

### Label

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `required` | `boolean` | `false` | When `true`, renders a red asterisk after the label text. |
| `htmlFor` | `string` | — | Associates the label with a form control by its `id`. |
| `className` | `string` | — | Additional CSS classes to merge. |

The `Label` component also accepts all standard HTML `<label>` attributes.
