### Manual

No dependencies required — copy and paste the following code into your project:

```tsx filename="components/ui/empty.tsx" copy
import * as React from "react"
import { cn } from "@/lib/utils"

function Empty({ className, ...props }: React.ComponentProps<"div">) {
  return (
    <div
      data-slot="empty"
      className={cn("flex flex-col items-center justify-center gap-6 py-12 text-center", className)}
      {...props}
    />
  )
}

function EmptyHeader({ className, ...props }: React.ComponentProps<"div">) {
  return (
    <div
      data-slot="empty-header"
      className={cn("flex flex-col items-center gap-3", className)}
      {...props}
    />
  )
}

function EmptyMedia({
  className,
  variant = "icon",
  ...props
}: React.ComponentProps<"div"> & { variant?: "icon" | "image" }) {
  return (
    <div
      data-slot="empty-media"
      data-variant={variant}
      className={cn(
        "flex items-center justify-center",
        variant === "icon" &&
          "bg-muted text-muted-foreground size-14 rounded-full [&_svg]:size-6 [&_svg]:shrink-0",
        variant === "image" && "overflow-hidden rounded-lg",
        className
      )}
      {...props}
    />
  )
}

function EmptyTitle({ className, ...props }: React.ComponentProps<"h3">) {
  return (
    <h3
      data-slot="empty-title"
      className={cn("text-base font-semibold", className)}
      {...props}
    />
  )
}

function EmptyDescription({ className, ...props }: React.ComponentProps<"p">) {
  return (
    <p
      data-slot="empty-description"
      className={cn("text-muted-foreground max-w-xs text-sm", className)}
      {...props}
    />
  )
}

function EmptyContent({ className, ...props }: React.ComponentProps<"div">) {
  return (
    <div
      data-slot="empty-content"
      className={cn("flex flex-col items-center gap-2", className)}
      {...props}
    />
  )
}

export { Empty, EmptyContent, EmptyDescription, EmptyHeader, EmptyMedia, EmptyTitle }
```

### Empty

Main container. Renders a centered, column-flexed `div`.

| Prop | Type | Description |
|------|------|-------------|
| `className` | `string` | Additional classes. |

### EmptyHeader

Groups `EmptyMedia`, `EmptyTitle`, and `EmptyDescription` together.

### EmptyMedia

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"icon" \| "image"` | `"icon"` | `"icon"` renders a muted rounded circle; `"image"` wraps with a rounded container. |
| `className` | `string` | — | Additional classes. |

### EmptyTitle

Renders an `<h3>` with a semibold style.

### EmptyDescription

Renders a `<p>` with `text-muted-foreground` and a max width constraint.

### EmptyContent

Wraps action elements (buttons, inputs, etc.) in a centered column.
